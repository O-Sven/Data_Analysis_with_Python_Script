def get_ball_position(filename):
    image = plt.imread(filename)#[:200, 1000:1200,:]
    r = image[:,:,0]
    g = image[:,:,1]
    b = image[:,:,2]

    r[r < 2*b] = 0
    r[r < 2*g] = 0
    r[r < 0.4] = 0

    x = 0
    y = 0
    n = 0
    for index, v in np.ndenumerate(r):
        if v > 0.01:
            y += index[0]
            x += index[1]
            n += 1
    
    return x/n, y/n
